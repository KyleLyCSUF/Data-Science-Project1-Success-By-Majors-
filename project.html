# ==== Feature engineering (no new imports) ====

# Helpful progress features
data['Grade_Difference'] = data['Curricular units 2nd sem (grade)'] - data['Curricular units 1st sem (grade)']
data['Avg_Grade'] = (data['Curricular units 1st sem (grade)'] + data['Curricular units 2nd sem (grade)']) / 2

# Robust approval ratio (avoid divide by zero)
data['Approval_Ratio'] = data['Curricular units 2nd sem (approved)'] / (data['Curricular units 2nd sem (enrolled)'] + 1)

# Optional evaluations intensity (also safe divide)
data['Eval_Intensity_1st'] = data['Curricular units 1st sem (evaluations)'] / (data['Curricular units 1st sem (enrolled)'] + 1)
data['Eval_Intensity_2nd'] = data['Curricular units 2nd sem (evaluations)'] / (data['Curricular units 2nd sem (enrolled)'] + 1)

# Log-transform GDP to stabilize variance
data['log_GDP'] = np.log1p(data['GDP'])

# Select features (all numeric, no new imports)
features = [
    'Unemployment rate', 'Inflation rate',
    'Admission grade', 'Age at enrollment',
    'Curricular units 1st sem (grade)', 'Curricular units 2nd sem (grade)',
    'Grade_Difference', 'Avg_Grade', 'Approval_Ratio',
    'Eval_Intensity_1st', 'Eval_Intensity_2nd'
]
target = 'log_GDP'

# Build X/y and drop any rows that became NaN/inf due to division
X = data[features].replace([np.inf, -np.inf], np.nan).dropna()
y = data.loc[X.index, target]

print(f"Modeling rows after cleaning: {len(X)}")

# ==== Train / evaluate improved linear model on log(GDP) ====

valid_idx = y.dropna().index
X = X.loc[valid_idx]
y = y.loc[valid_idx]

# Train/test split
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.20, random_state=42
)

# Scale features (you already imported StandardScaler)
scaler = StandardScaler()
X_train_s = scaler.fit_transform(X_train)
X_test_s  = scaler.transform(X_test)

# Fit multiple linear regression (you already imported LinearRegression)
linreg = LinearRegression()
linreg.fit(X_train_s, y_train)

# Predict log(GDP), then bring back to original GDP scale
y_pred_log = linreg.predict(X_test_s)
y_pred = np.expm1(y_pred_log)          # inverse of log1p
y_true = np.expm1(y_test.to_numpy())

# Metrics on original scale
r2_orig = r2_score(y_true, y_pred)
mse_orig = mean_squared_error(y_true, y_pred)
mae_orig = np.mean(np.abs(y_true - y_pred))

print(f"R (original GDP): {r2_orig:.3f}")
print(f"MSE (original GDP): {mse_orig:.3f}")
print(f"MAE (original GDP): {mae_orig:.3f}")

# Coefficients (still in log-target space; larger magnitude => stronger effect)
coef_df = pd.DataFrame({
    'Feature': features,
    'Coefficient_on_logGDP': linreg.coef_
}).sort_values('Coefficient_on_logGDP', ascending=False)
display(coef_df)
